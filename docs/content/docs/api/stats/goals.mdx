## Get Goals

<div className="grid grid-cols-1 xl:grid-cols-2 gap-8">
<div>

```
GET /api/goals/:site
```

Returns a paginated list of goals with their conversion metrics.

### Path Parameters

<TypeTable
  type={{
    site: {
      description: 'Site ID or site identifier',
      type: 'string | number',
      required: true,
    },
  }}
/>

### Query Parameters

Accepts all [Common Parameters](/docs/api/stats/reference#common-parameters) plus the following:

<TypeTable
  type={{
    page: {
      description: 'Page number (1-indexed)',
      type: 'number',
      default: 1,
    },
    page_size: {
      description: 'Number of results per page (max 100)',
      type: 'number',
      default: 10,
    },
    sort: {
      description: 'Field to sort by',
      type: 'string',
      typeDescription: '"goalId" | "name" | "goalType" | "createdAt"',
      default: 'createdAt',
    },
    order: {
      description: 'Sort direction',
      type: 'string',
      typeDescription: '"asc" | "desc"',
      default: 'desc',
    },
  }}
/>

### Response

<TypeTable
  type={{
    data: {
      description: 'Array of goal objects with conversion data',
      type: 'Goal[]',
    },
    meta: {
      description: 'Pagination metadata',
      type: 'PaginationMeta',
    },
  }}
/>

#### Goal Object

<TypeTable
  type={{
    goalId: {
      description: 'Unique goal identifier',
      type: 'number',
    },
    name: {
      description: 'Goal name',
      type: 'string | null',
    },
    goalType: {
      description: 'Type of goal',
      type: 'string',
      typeDescription: '"path" | "event"',
    },
    config: {
      description: 'Goal configuration',
      type: 'GoalConfig',
    },
    createdAt: {
      description: 'Creation timestamp',
      type: 'string',
      typeDescription: 'ISO 8601 datetime',
    },
    total_conversions: {
      description: 'Number of sessions that completed the goal',
      type: 'number',
    },
    total_sessions: {
      description: 'Total sessions in the time period',
      type: 'number',
    },
    conversion_rate: {
      description: 'Conversion rate (0-1)',
      type: 'number',
    },
  }}
/>

#### GoalConfig Object

<TypeTable
  type={{
    pathPattern: {
      description: 'URL path pattern (for path goals). Supports wildcards (*)',
      type: 'string',
      required: false,
    },
    eventName: {
      description: 'Event name to match (for event goals)',
      type: 'string',
      required: false,
    },
    eventPropertyKey: {
      description: 'Event property key to match',
      type: 'string',
      required: false,
    },
    eventPropertyValue: {
      description: 'Event property value to match',
      type: 'string | number | boolean',
      required: false,
    },
  }}
/>

## Create Goal

<div className="grid grid-cols-1 xl:grid-cols-2 gap-8">
<div>

```
POST /api/goal/create
```

Creates a new goal for tracking conversions.

### Request Body

<TypeTable
  type={{
    siteId: {
      description: 'Site ID to create the goal for',
      type: 'number',
      required: true,
    },
    name: {
      description: 'Goal name',
      type: 'string',
      required: false,
    },
    goalType: {
      description: 'Type of goal',
      type: 'string',
      typeDescription: '"path" | "event"',
      required: true,
    },
    config: {
      description: 'Goal configuration',
      type: 'GoalConfig',
      required: true,
    },
  }}
/>

#### GoalConfig for Path Goals

<TypeTable
  type={{
    pathPattern: {
      description: 'URL path pattern to match. Supports wildcards (*)',
      type: 'string',
      required: true,
      typeDescription: 'e.g., "/checkout/*", "/signup/complete"',
    },
  }}
/>

#### GoalConfig for Event Goals

<TypeTable
  type={{
    eventName: {
      description: 'Event name to match',
      type: 'string',
      required: true,
    },
    eventPropertyKey: {
      description: 'Property key to match (optional)',
      type: 'string',
      required: false,
    },
    eventPropertyValue: {
      description: 'Property value to match (required if key is provided)',
      type: 'string | number | boolean',
      required: false,
    },
  }}
/>

### Response

<TypeTable
  type={{
    success: {
      description: 'Whether the goal was created',
      type: 'boolean',
    },
    goalId: {
      description: 'ID of the created goal',
      type: 'number',
    },
  }}
/>

</div>
<div className="flex flex-col gap-4">

<Tabs items={['cURL', 'JavaScript', 'Python']}>
  <Tab value="cURL">
```bash title="Path Goal"
curl -X POST "https://api.rybbit.io/api/goal/create" \
  -H "Authorization: Bearer your_api_key_here" \
  -H "Content-Type: application/json" \
  -d '{
    "siteId": 123,
    "name": "Checkout Complete",
    "goalType": "path",
    "config": {
      "pathPattern": "/checkout/success"
    }
  }'
```
  </Tab>
  <Tab value="JavaScript">
```javascript title="Event Goal"
const response = await fetch(
  'https://api.rybbit.io/api/goal/create',
  {
    method: 'POST',
    headers: {
      'Authorization': 'Bearer your_api_key_here',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      siteId: 123,
      name: 'Purchase Made',
      goalType: 'event',
      config: {
        eventName: 'purchase',
        eventPropertyKey: 'plan',
        eventPropertyValue: 'pro'
      }
    })
  }
);

const data = await response.json();
```
  </Tab>
  <Tab value="Python">
```python title="Request"
import requests

response = requests.post(
    'https://api.rybbit.io/api/goal/create',
    json={
        'siteId': 123,
        'name': 'Signup Complete',
        'goalType': 'path',
        'config': {
            'pathPattern': '/signup/*'
        }
    },
    headers={
        'Authorization': 'Bearer your_api_key_here'
    }
)

data = response.json()
```
  </Tab>
</Tabs>

```json title="Response"
{
  "success": true,
  "goalId": 3
}
```

</div>
</div>
